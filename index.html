  Game Over! <span id="final-score"></span>

  <button id="retry-button">Retry</button>

</div>

<p>スコア: <span id="score">0</span></p>


<script>

  let circles = [];

  let gameInterval;

  let score = 0;


  class Circle {

      constructor(x, y, color) {

          this.element = document.createElement('div');

          this.element.className = 'circle';

          this.element.style.backgroundColor = color;

          this.setPosition(x, y);

      }

      setPosition(x, y) {

          this.x = x;

          this.y = y;

          this.element.style.left = x + 'px';

          this.element.style.top = y + 'px';

      }

      updatePosition() {

          this.x += (Math.random() - 0.5) * 5;

          this.y += (Math.random() - 0.5) * 5;

          this.setPosition(this.x, this.y);

      }

  }


  function updateGame() {

      circles.forEach(circle => circle.updatePosition());

  }


  function gameOver() {

      clearInterval(gameInterval);

      const gameOverScreen = document.getElementById('game-over-screen');

      gameOverScreen.classList.remove('hidden');

      document.getElementById('final-score').textContent = score;

  }


  function startGame() {

      document.getElementById('start-screen').classList.add('hidden');

      score = 0;

      document.getElementById('score').textContent = score;


      const gameArea = document.getElementById('game-area');

      gameArea.innerHTML = '';

      circles = [];


      const rect = gameArea.getBoundingClientRect();

      const initialCircle = new Circle(rect.width / 2, rect.height / 2, 'white');

      gameArea.appendChild(initialCircle.element);

      circles.push(initialCircle);


      gameInterval = setInterval(updateGame, 16);

  }


  document.addEventListener('DOMContentLoaded', function() {

      const startScreen = document.getElementById('start-screen');

      startScreen.addEventListener('click', function(e) {

          e.stopPropagation();

          startGame();

      });

      startScreen.addEventListener('touchstart', function(e) {

          e.stopPropagation();

          startGame();

      });


      const retryButton = document.getElementById('retry-button');

      retryButton.addEventListener('click', function(e) {

          e.stopPropagation();

          startGame();

      });


      const gameArea = document.getElementById('game-area');

      gameArea.addEventListener('click', function(e) {

          e.stopPropagation();

      });

      gameArea.addEventListener('touchstart', function(e) {

          e.stopPropagation();

      });

  });

</script>
